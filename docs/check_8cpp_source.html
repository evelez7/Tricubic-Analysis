176 double error_calculation(std::shared_ptr<std::list<double>> const &approximate_list,
177  std::shared_ptr<std::list<double>> const &exact_list) {
178  auto difference_list = std::make_shared<std::list<double>>();
179 
180  if (approximate_list->size() != exact_list->size()) {
181  throw "Vectors not of equal size";
182  }
183 
184  auto approx_it = approximate_list->begin();
185  auto exact_it = exact_list->begin();
186 
187  for(; approx_it != approximate_list->end() && exact_it != exact_list->end(); ++approx_it, ++exact_it) {
188  difference_list->push_back((*exact_it) - (*approx_it));
189  }
190 
191  auto difference_norm = norm(difference_list);
192  auto difference_norm_squared = pow(difference_norm, 2.0);
193 
194  auto exact_norm = norm(exact_list);
195  auto exact_norm_squared = pow(exact_norm, 2.0);
196 
197  auto final = sqrt(difference_norm_squared / exact_norm_squared);
198  return final;
199 }
